# Config
$resourceGroup = "functions-msp"
$appName = "voicelauncherblazor"
$dbPath = "C:\Users\MPhil\source\repos\VoiceLauncherBlazor\voicelauncher-azure.db"
$targetPath = "site/wwwroot/voicelauncher-azure.db"

# Get publishing profile (deployment credentials)
$profile = az webapp deployment list-publishing-profiles `
    --name $appName `
    --resource-group $resourceGroup `
    | ConvertFrom-Json

$publishProfile = $profile[0]

$username = $publishProfile.userName
$password = $publishProfile.userPWD

# Build auth header
$pair = "$username`:$password"
$bytes = [System.Text.Encoding]::ASCII.GetBytes($pair)
$encoded = [Convert]::ToBase64String($bytes)

$headers = @{
    Authorization = "Basic $encoded"
}

# Upload file
Invoke-RestMethod `
    -Uri "https://$appName.scm.azurewebsites.net/api/vfs/$targetPath" `
    -Headers $headers `
    -Method PUT `
    -InFile $dbPath `
    -ContentType "application/octet-stream"
